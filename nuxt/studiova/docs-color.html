Color

Override Light Theme Colors

For any change in colors : theme/LightTheme.ts 
                        
const LIGHT_THEME: ThemeTypes = {
    name: 'LIGHT_THEME',
    dark: false,
    variables: {
        'border-color': '#e0e6eb',
        'border-opacity': 1,
    },
    colors: {
        primary: '#C1FF72',
        secondary: '#1f2a2e',
        muted:'#E0E0E0',
        white:'#ffffff',
        lightgray:'#F4F8FA',
        dark:'#1f2a2e',
        surface:'#F4F8FA',
        accent:'#ffffff'
    }
};

Override Dark Theme Colors

For any change in colors : theme/DarkTheme.ts 
                        
const DARK_THEME: ThemeTypes = {
    name: 'DARK_THEME',
    dark: true, 
    variables: {
        'border-color': '#e0e6eb',
        'border-opacity': 0.2,
    },
   
    colors: {
        primary: '#C1FF72',
        secondary: '#1f2a2e',
        muted:'#1f2a2eb3',
        white:'#ffffff',
        lightgray:'#273338',
        dark:'#ffffff',
        surface:'#1F2A2E',
        darkgray:'#273338',
        accent:'#343F42'
    }
};
